\begin{frame}{}
  \begin{theorem}[概括原则]
    对于任意性质/谓词 $P(x)$, 都存在一个集合 $X$:
    \[
      X = \set{x \mid P(x)}
    \]
  \end{theorem}

  \pause
  \begin{definition}[Russell's Paradox]
    \[
      \blue{P(x) \triangleq ``x \notin x"}
    \]

    \pause
    \[
      R = \set{x \mid x \notin x}
    \]

    \pause
    \[
      \red{Q: R \in R\;?}
    \]
  \end{definition}
\end{frame}
