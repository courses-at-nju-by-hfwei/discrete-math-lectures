\begin{frame}{}
  \begin{theorem}[概括原则]
    \blue{For any predicate $\psi(x)$}, there is a \red{set} $X$:
    \[
      X = \set{x \mid \psi(x)}.
    \]
  \end{theorem}

  \pause
  \begin{definition}[Russell's Paradox]
    \[
      \blue{\psi(x) \triangleq ``x \notin x"}
    \]

    \pause
    \[
      R = \set{x \mid x \notin x}
    \]

    \pause
    \[
      \red{Q: R \in R\;?}
    \]
  \end{definition}
\end{frame}
