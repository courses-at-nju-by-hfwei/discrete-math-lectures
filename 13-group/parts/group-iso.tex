\begin{frame}
  \begin{definition}[Isomorphism (同构)]
    Let $(G, \cdot)$ and $(G', \ast)$ be two groups.
    Let $\phi$ be a \blue{bijection} such that
    \[
      \forall a, b \in G.\; \phi(a \cdot b) = \phi(a) \ast \phi(b).
    \]
    Then $\phi$ is an \red{isomorphism} from $G$ to $G'$.
  \end{definition}

  \pause
  \vspace{0.50cm}
  \begin{center}
    $G$ and $G'$ are isomorphic
    \[
      \phi: G \cong G'
    \]
  \end{center}
\end{frame}

\begin{frame}{}
  \begin{theorem}
    Suppose that \blue{$\phi: G \cong G'$}.
    Let $e$ and $e'$ are identities of $G$ and $G'$, respectively.

    \begin{enumerate}[(1)]
      \setlength{\itemsep}{6pt}
      \item $\phi(e) = e'$
      \item $\phi(a^{-1}) = (\phi(a))^{-1}$
      \item $\phi^{-1}: G' \cong G$
    \end{enumerate}
  \end{theorem}
\end{frame}
