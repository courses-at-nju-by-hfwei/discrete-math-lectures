    To \blue{maximize} the size of a mathematical structure $\mathcal{S}$ in $G$

    \pause

    To \blue{minimize} the size of its \red{dual} mathematical structure $\mathcal{S}'$ in $G$

    \pause
    \vspace{0.30cm}
    \begin{theorem}[Weak Duality Theorem]
      The size of a maximum $\mathcal{S}$ \red{$\le$} The size of a minimum $\mathcal{S'}$
    \end{theorem}

    \pause
    \vspace{0.30cm}
    \begin{theorem}[Strong Duality Theorem]
      The size of a maximum $\mathcal{S}$ \red{=} The size of a minimum $\mathcal{S'}$
    \end{theorem}
