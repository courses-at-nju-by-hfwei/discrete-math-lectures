\begin{frame}{}
  \begin{theorem}[Cut Property]
    \begin{description}
      \item[$X:$] A part of some MST $T$ of $G$
      \item[$(S, V \setminus S):$] A \purple{\it cut} such that $X$ does \purple{\it not} cross $(S, V \setminus S)$
    ­  \item[$e:$] \red{A} lightest edge across $(S, V \setminus S)$
    \end{description}

    \begin{center}
      {Then \blue{$X \cup \set{e}$} is a part of \red{some} MST $T'$ of $G$.}
    \end{center}
  \end{theorem}
\end{frame}
