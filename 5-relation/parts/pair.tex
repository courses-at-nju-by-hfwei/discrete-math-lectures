\begin{frame}{}
  \begin{definition}[$n$-元组 (n-ary tuples)]
    \[
      (x, y, z) \triangleq ((x, y), z)
    \]

    \pause
    \[
      (x_{1}, x_{2}, \dots, x_{n-1}, x_{n})
        \triangleq ((x_{1}, x_{2}, \dots, x_{n-1}), x_{n})
    \]
  \end{definition}

  \pause
  \vspace{0.60cm}
  \begin{theorem}
    \[
      (x_{1}, \dots, x_{n}) = (y_{1}, \dots, y_{n})
      \iff x_{1} = y_{1} \land \dots x_{n} = y_{n}
    \]
  \end{theorem}
\end{frame}
