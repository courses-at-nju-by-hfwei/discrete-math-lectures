\begin{frame}{}
  \begin{definition}[Cartesian Products]
    The \red{\it Cartesian product} $A \times B$ of $A$ and $B$ is defined as 
    \[
      {A \times B \triangleq \set{(a, b) \mid a \in A \land b \in B}}
    \]
  \end{definition}

  \pause

    X^2 \triangleq X \times X
