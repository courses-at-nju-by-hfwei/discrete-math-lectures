\begin{frame}{}
  \begin{definition}{}
    If $G$ is a \teal{connected} undirected graph \teal{without loops},
    then $G$ is \red{$k$-colorable} if its vertices can be colored
    in \purple{$\le k$} colors so that adjacent vertices have different colors.
  \end{definition}
\end{frame}
